學院、系所、學位、年級 Vue components

# 基本設定


  import Vue from 'vue'
  import components from 'vue-ncu-unit'

  new Vue({
    el: '#app',
    components: {
      college: components.College,
      degree: components.Degree,
      department: components.Department,
      grade: components.Grade,
      trans: components.Trans
    }
    // or in es2015...
    components
  })

# 事件

demoCallback: 


import { bus, events } from '../'

let event = events.changeDegree.withID('demo-degree')
// event = 'changeDegree:demo-degree'

bus.$on(event, function (data) {
  document.getElementById('demoCallback').textContent = data.degree.id
})

# 參數篩選系所

排除 except 僅剩 only

若ID同時出現在except和only，則先作except再作only


<department defaults="32060"
           :filtering="{
              college: '工學院',
              degree: '碩士班'
            }"
           :except="[32010, 32020, 32030, 32040]"></department>

<department defaults="32060"
            :filtering="{
              college: '工學院',
              degree: '碩士班'
            }"
            :only="[32060, 32080, 32090]">

SQL

可將系所、學院資料匯入後端資料庫，以便其他運用。

# 系所


CREATE TABLE `departments` (
    `id` varchar(10) COLLATE utf8mb4_unicode_ci NOT NULL,
    `group_id` int(1) NOT NULL COMMENT '系所組代碼',
    `chinese_name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '中文名稱',
    `english_name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '英文名稱',
    `study_system_no` int(1) NOT NULL COMMENT '學級代碼',
    `parent_id` varchar(10) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '本系所組屬於哪個系所',
    `college_id` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '屬於哪個學院',
    `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    `deleted_at` int(11) DEFAULT NULL,
    PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='系所';


# 學院


CREATE TABLE `colleges` (
  `id` varchar(50) COLLATE utf8mb4_unicode_ci NOT NULL,
  `chinese_name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '中文名稱',
  `english_name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '英文名稱',
  `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `deleted_at` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='學院';
